⚙ Definições
Configure a integração com o Google Sheets e as preferências do sistema — sem editar o código HTML.
📂 A usar armazenamento local. Configure o Google Sheets abaixo.
1
Criar o Google Sheet
Aceda a sheets.google.com e crie uma folha nova. O nome da folha será Encomendas — o script cria-a automaticamente se não existir.
2
Abrir o Apps Script e colar o código
No Google Sheet vá a Extensões → Apps Script. Apague todo o conteúdo que lá estiver e cole o código abaixo. Depois clique em Guardar (Ctrl+S).
// ─────────────────────────────────────────────
// Fly Way Logistics — Google Apps Script
// Cole este código em Extensões → Apps Script
// ─────────────────────────────────────────────

const SHEET_NAME = 'Encomendas';
const COLS = ['id','name','phone','product','origin','dest','weight','status','updatedAt','history'];

function getSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) { sh = ss.insertSheet(SHEET_NAME); sh.appendRow(COLS); }
  if (sh.getLastRow() === 0) sh.appendRow(COLS);
  return sh;
}

function doGet(e) {
  const p = e.parameter;
  const sh = getSheet();
  let result;

  if (p.action === 'getAll') {
    const rows = sh.getDataRange().getValues();
    if (rows.length <= 1) { result = { ok: true, orders: [] }; }
    else {
      const orders = rows.slice(1).map(r => {
        const o = {};
        COLS.forEach((h, i) => o[h] = r[i]);
        try { o.history = JSON.parse(o.history || '[]'); } catch(e2) { o.history = []; }
        return o;
      });
      result = { ok: true, orders };
    }

  } else if (p.action === 'create') {
    const o = JSON.parse(decodeURIComponent(p.data));
    sh.appendRow(COLS.map(h => h === 'history' ? JSON.stringify(o[h]||[]) : (o[h]||'')));
    result = { ok: true };

  } else if (p.action === 'update') {
    const o = JSON.parse(decodeURIComponent(p.data));
    const vals = sh.getDataRange().getValues();
    for (let i = 1; i < vals.length; i++) {
      if (vals[i][0] === o.id) {
        sh.getRange(i+1, 1, 1, COLS.length).setValues([
          COLS.map(h => h === 'history' ? JSON.stringify(o[h]||[]) : (o[h]||''))
        ]);
        break;
      }
    }
    result = { ok: true };

  } else if (p.action === 'delete') {
    const vals = sh.getDataRange().getValues();
    for (let i = 1; i < vals.length; i++) {
      if (vals[i][0] === p.id) { sh.deleteRow(i+1); break; }
    }
    result = { ok: true };

  } else {
    result = { ok: false, error: 'Ação desconhecida: ' + p.action };
  }

  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

Publicar como Web App
